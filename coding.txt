void gotoxy(int a,int b)
{
 COORD x;
 x.X=a;
 x.Y=b;
 SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE),x);


}
int facto(int a)
{
   int f,i;
   f=1;
   for(i=0;i<a;i++)
   {
     f=f*(i+1);

   }
   return f;



}
int factore(int a)
{
    if(a==1)
    return 1;
    else
    return (a*factore(a-1));




}
void readfile(void)
{

    FILE *p;
    char ch;
    p=fopen("coding.txt","r");
    if(p==NULL)
    {
        printf("FILE NOT FOUND");
        getch();
        main();

    }
    while(!feof(p))
    {
        printf("%c",ch);
        ch=fgetc(p);

    }
    fclose(p);



}
